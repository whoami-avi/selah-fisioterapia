2) Esquema (DDL): tablas, PK/FK, restricciones, indices

Diagrama de relaciones

users (autenticacion)   |   patients (1) ——< appointments (N)   |   templates   |   messages

DDL - Tabla users

-- Usuarios del sistema (admin/asistente)
CREATE TABLE public.users (
    id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    username      VARCHAR(50) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    role          VARCHAR(20) NOT NULL DEFAULT 'asistente'
                  CHECK (role IN ('admin', 'asistente')),
    name          VARCHAR(100) NOT NULL,
    email         VARCHAR(100),
    active        BOOLEAN DEFAULT true,
    created_at    TIMESTAMPTZ DEFAULT now(),
    updated_at    TIMESTAMPTZ DEFAULT now()
);
CREATE INDEX idx_users_username ON public.users(username);

DDL - Tabla patients

-- Pacientes de la clinica
CREATE TABLE public.patients (
    id                UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name              VARCHAR(150) NOT NULL,
    phone             VARCHAR(20) NOT NULL,
    email             VARCHAR(100),
    birthdate         DATE,
    occupation        VARCHAR(100),
    referrer          VARCHAR(100),
    address           TEXT,
    diagnosis         TEXT,
    treatment         VARCHAR(100),
    notes             TEXT,
    emergency_contact VARCHAR(100),
    emergency_phone   VARCHAR(20),
    created_at        TIMESTAMPTZ DEFAULT now(),
    updated_at        TIMESTAMPTZ DEFAULT now()
);
CREATE INDEX idx_patients_name ON public.patients(name);
CREATE INDEX idx_patients_created ON public.patients(created_at DESC);

DDL - Tabla appointments (FK → patients)

-- Citas medicas vinculadas a pacientes
CREATE TABLE public.appointments (
    id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    patient_id    UUID NOT NULL REFERENCES public.patients(id)
                  ON DELETE CASCADE,
    date          DATE NOT NULL,
    time          VARCHAR(5) NOT NULL,
    duration      INTEGER DEFAULT 60,
    type          VARCHAR(100),
    status        VARCHAR(20) DEFAULT 'scheduled'
                  CHECK (status IN ('scheduled','confirmed',
                        'completed','cancelled','pending')),
    notes         TEXT,
    costo         NUMERIC(10,2) DEFAULT 0,
    metodo_pago   VARCHAR(20)
                  CHECK (metodo_pago IN ('efectivo','tarjeta','transferencia')),
    pago_estado   VARCHAR(20) DEFAULT 'pendiente'
                  CHECK (pago_estado IN ('pagado','pendiente','parcial')),
    created_at    TIMESTAMPTZ DEFAULT now(),
    updated_at    TIMESTAMPTZ DEFAULT now()
);
CREATE INDEX idx_appointments_date ON public.appointments(date);
CREATE INDEX idx_appointments_patient ON public.appointments(patient_id);
CREATE INDEX idx_appointments_date_status ON public.appointments(date, status);

4) Consultas requeridas

Q1 — JOIN principal: Agenda con datos de paciente

Caso de uso: Mostrar la agenda del dia con nombre, telefono y diagnostico de cada paciente.

SELECT
    a.date            AS fecha,
    a.time            AS hora,
    p.name            AS paciente,
    p.phone           AS telefono,
    p.diagnosis       AS diagnostico,
    a.type            AS tipo_terapia,
    a.status          AS estado,
    a.costo
FROM appointments a
INNER JOIN patients p ON a.patient_id = p.id
WHERE a.date = '2026-04-18'
ORDER BY a.time ASC;

Q2 — Filtro estado/fecha: Citas pagadas del mes

Caso de uso: Reporte financiero de pagos confirmados en abril.

SELECT
    a.date            AS fecha,
    p.name            AS paciente,
    a.type            AS tipo_terapia,
    a.costo,
    a.metodo_pago
FROM appointments a
INNER JOIN patients p ON a.patient_id = p.id
WHERE a.pago_estado = 'pagado'
  AND a.date BETWEEN '2026-04-01' AND '2026-04-30'
ORDER BY a.date;

Q3 — Reporte/resumen: Ingresos por metodo de pago

Caso de uso: Dashboard financiero con totales agrupados.

SELECT
    metodo_pago,
    COUNT(*)              AS total_citas,
    SUM(costo)            AS ingresos_totales,
    ROUND(AVG(costo), 2)  AS promedio_por_cita
FROM appointments
WHERE pago_estado = 'pagado'
GROUP BY metodo_pago
ORDER BY ingresos_totales DESC;

Snippet de conexion

// Dependencia CDN
<script src="https://cdn.jsdelivr.net/npm/@supabase/supabase-js@2"></script>

// Inicializacion del cliente
const SUPABASE_URL = 'https://uomwyiapknnplqxmglnv.supabase.co';
const SUPABASE_KEY = 'eyJhbGciOiJIUzI1NiIs...';

let supabaseClient = null;

function initSupabase() {
    if (SUPABASE_URL && SUPABASE_KEY && window.supabase) {
        supabaseClient = window.supabase.createClient(
            SUPABASE_URL, SUPABASE_KEY
        );
        console.log('Supabase client inicializado');
        return true;
    }
    return false;
}

// Verificacion de conexion
async function testConnection() {
    const { error } = await supabaseClient
        .from('patients')
        .select('count', { count: 'exact', head: true });
    return !error; // true = conectado
}

// JavaScript (Supabase SDK)
const { data, error } = await supabaseClient
    .from('patients')
    .select('*')
    .order('created_at', { ascending: false });

-- SQL equivalente:
SELECT * FROM patients ORDER BY created_at DESC;
Resultado: 4 pacientes obtenidos correctamente desde Supabase.

INSERT — Registrar nuevo paciente

        name: 'Ana Martinez',

Resultado: Paciente insertado con UUID autogenerado.

UPDATE — Actualizar notas del paciente

const { data, error } = await supabaseClient
    .from('patients')
    .update({ notes: 'Responde bien al tratamiento' })
    .eq('id', 'cf677621-56c3-40b4-8c23-b3eea0547f1c')
    .select().single();

-- SQL equivalente:
UPDATE patients
SET notes = 'Responde bien al tratamiento'
WHERE id = 'cf677621-...'
RETURNING *;
Resultado: Campo notes actualizado para Juan Perez.

DELETE — Eliminar paciente (CASCADE a citas)

const { error } = await supabaseClient
    .from('patients')
    .delete()
    .eq('id', 'fdd73884-c853-435c-abec-2cdad94e6349');

-- SQL equivalente:
DELETE FROM patients
WHERE id = 'fdd73884-...';
-- Citas asociadas se eliminan por ON DELETE CASCADE
Resultado: Paciente eliminado junto con citas asociadas.
